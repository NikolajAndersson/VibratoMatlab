 74

 76        plugin.Mix = val;
 77        paramState.Mix = val;
 78        checkForTampering(plugin, paramState, sampleRate, ...
 79            'Setting parameter ''Mix''');
 80        
 81        % Get a frame of input data
 82        in = ibuf(iptr:iptr+samplesPerFrame-1, :);
 83        iptr = iptr + samplesPerFrame;
 84        
 85        % Run the plug-in

 87        
 88        % Save the output data if requested

123function checkForTampering(plugin, paramState, sampleRate, cause)
124% Verify parameters were not tampered with
125if ~isequal(paramState.width, plugin.width)
126    error('ValidateAudioPlugin:ParamChanged', ...
127        '%s changed parameter ''width'' from %g to %g.', ...
128        cause, paramState.width, plugin.width);
129end
130if ~isequal(paramState.rate, plugin.rate)
131    error('ValidateAudioPlugin:ParamChanged', ...
132        '%s changed parameter ''rate'' from %g to %g.', ...
133        cause, paramState.rate, plugin.rate);
134end
135if ~isequal(paramState.waveform, plugin.waveform)
136    error('ValidateAudioPlugin:ParamChanged', ...
137        '%s changed parameter ''waveform'' from ''%s'' to ''%s'.'', ...
138        cause, paramState.waveform, plugin.waveform);
139end
140if ~isequal(paramState.Mix, plugin.Mix)
141    error('ValidateAudioPlugin:ParamChanged', ...
142        '%s changed parameter ''Mix'' from %g to %g.', ...
143        cause, paramState.Mix, plugin.Mix);
144end
145% Verify sample rate was not tampered with
146if ~isequal(getSampleRate(plugin), sampleRate)
147    error('ValidateAudioPlugin:SampleRateChanged', ...
148        '%s changed sample rate from %g to %g.', ...
149        cause, sampleRate, getSampleRate(plugin));
150end
151end

153function y = logchirp(f0, f1, Fs, nsamples, initialPhase)
154% logarithmically swept sine from f0 to f1 over nsamples, at Fs
155y = zeros(nsamples,numel(initialPhase));
156instPhi = logInstantaneousPhase(f0, f1, Fs, nsamples);
157for i = 1:numel(initialPhase)
158    y(:,i) = sin(instPhi + initialPhase(i));
159end
160end
161
162function phi = logInstantaneousPhase(f0, f1, Fs, n)
163final = n-1;
164t = (0:final)/final;
165t1 = final/Fs;
166phi = 2*pi * t1/log(f1/f0) * (f0 * (f1/f0).^(t') - f0);
167end

169function paramState = initParamState(plugin)
170paramState.width = plugin.width;
171paramState.rate = plugin.rate;
172paramState.waveform = plugin.waveform;
173paramState.Mix = plugin.Mix;
174end

176function val = fromNormalizedwidth(normval)
177val = 1e-05 + (0.001-1e-05)*normval;
178end
179
180function val = fromNormalizedrate(normval)
181val = 0.1 + (14-0.1)*normval;
182end
183
184function val = fromNormalizedwaveform(normval)
185idx = floor( 0.5 + normval * 3);
186switch idx
187    case 0
188        val = 'Sine';
189    case 1
190        val = 'Square';
191    case 2
192        val = 'Sawtooth';
193    otherwise
194        val = 'Triangle';
195end
196end
197
198function val = fromNormalizedMix(normval)
199val = 0 + (1-0)*normval;
200end
